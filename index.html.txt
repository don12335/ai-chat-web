<body>
  <div class="container">
    <h2>🧠 AI 聊天機器人</h2>
    <div id="chat"></div>
    <input id="userInput" type="text" placeholder="輸入訊息..." />
    <button onclick="sendMessage()">送出</button>
  </div>

  <script>
    const chat = document.getElementById('chat');
    const input = document.getElementById('userInput');

    const API_KEY = '你的 OpenAI API 金鑰'; // ⚠️ 請替換成你的金鑰！

    async function sendMessage() {
      const userMessage = input.value;
      if (!userMessage.trim()) return;

      appendMessage('你', userMessage, 'user');
      input.value = '';
      
      appendMessage('AI', '思考中...', 'ai');

      const response = await fetch('https://api.openai.com/v1/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${API_KEY}`,
        },
        body: JSON.stringify({
          model: 'gpt-3.5-turbo',
          messages: [{ role: 'user', content: userMessage }],
        }),
      });

      const data = await response.json();
      const aiReply = data.choices?.[0]?.message?.content?.trim() || 'AI 沒有回應。';
      
      // 移除 “思考中...” 的訊息
      chat.removeChild(chat.lastChild);
      appendMessage('AI', aiReply, 'ai');
    }

    function appendMessage(sender, text, className) {
      const div = document.createElement('div');
      div.className = `message ${className}`;
      div.textContent = `${sender}：${text}`;
      chat.appendChild(div);
      chat.scrollTop = chat.scrollHeight;
    }
  </script>
